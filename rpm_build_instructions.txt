# 1. Установите инструменты для сборки RPM
sudo dnf install zlib-ng-compat-devel brotli-devel systemd-devel libselinux-devel cmake gcc-c++ rpm-build pkgconf-pkg-config

# 2. Инициализируйте дерево сборки (если не сделано ранее)
rpmdev-setuptree

# 3. Сделайте скрипт исполняемым
chmod +x build-rpm.sh

# 4. Запустите сборку
./build-rpm.sh 1.0.0

# 5. Установите собранный пакет
sudo dnf install ~/rpmbuild/RPMS/x86_64/mh-compressor-manager-1.0.0-1*.rpm

# 6. Проверьте установку
rpm -qi mh-compressor-manager
systemctl status mh-compressor-manager

# Проверка SPEC-файла на ошибки
rpmlint mh-compressor-manager.spec

# Сборка SRPM (исходный RPM)
rpmbuild -bs mh-compressor-manager.spec

# Установка из локального RPM
sudo dnf install ./mh-compressor-manager-1.0.0-1.x86_64.rpm

# Удаление пакета
sudo dnf remove mh-compressor-manager

# Проверка содержимого пакета
rpm -ql mh-compressor-manager
